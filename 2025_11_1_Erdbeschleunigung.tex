	\section*{Theoretischer Hintergrund}
	
	Zu den Bewegungen mit konstanter Beschleunigung gehört auch die des \textbf{freien Falls}. Das ist eine Bewegung, die entsteht, wenn auf einen fallenden Körper nur seine Gewichtskraft einwirkt. Die Fallbeschleunigung wird als Größe mit $g$ bezeichnet.
	
	\begin{formelbox}[title={\faIcon{square-root-alt} Einheit der Fallbeschleunigung}]
		\begin{equation*}
			[g] = \si{\metre\per\second\squared}
		\end{equation*}
	\end{formelbox}
	
	\begin{infobox}[title={\faIcon{info-circle} Ortsabhängigkeit der Erdbeschleunigung}]
		Die Fallbeschleunigung oder Erdbeschleunigung ist \textbf{keine Naturkonstante}. Mit Hilfe des Gravitationsgesetzes (Massenanziehung) kann diese Größe für jeden Ort bestimmt werden. Auf der Erde spielt dabei der Abstand zum Erdmittelpunkt eine wesentliche Rolle.
		
		\vspace{0.3cm}
		\begin{center}
			\begin{tabular}{lc}
				\toprule
				\textbf{Ort} & \textbf{Fallbeschleunigung} \\
				\midrule
				Panama & \SI{9,78}{\metre\per\second\squared} \\
				Nordpol & \SI{9,83}{\metre\per\second\squared} \\
				Jupiter & ca. \SI{26,00}{\metre\per\second\squared} \\
				\bottomrule
			\end{tabular}
		\end{center}
	\end{infobox}
	
	Wenn man Fallwege und Fallzeiten misst, so gelten die Gesetze für die Bewegung mit konstanter Beschleunigung:
	
	\begin{formelbox}[title={\faIcon{square-root-alt} Bewegungsgleichungen des freien Falls}]
		\textbf{Weg-Zeit-Gesetz:}
		\begin{equation*}
			s = \frac{1}{2} \cdot a \cdot t^2 \quad \Rightarrow \quad h = \frac{1}{2} \cdot g \cdot t^2
		\end{equation*}
		
		\textbf{Geschwindigkeit-Zeit-Gesetz:}
		\begin{equation*}
			v = a \cdot t \quad \Rightarrow \quad v = g \cdot t
		\end{equation*}
	\end{formelbox}

	\section*{Versuchsprinzip}
	
	Um das Weg-Zeit-Gesetz anzuwenden, lässt man eine Kugel, die von einem Elektromagneten gehalten wird, Höhen zwischen \SI{0,1}{\metre} und \SI{1,0}{\metre} durchfallen. 
	
	\begin{hilfebox}
		Der Elektromagnet wird mit einer \textbf{Wechselspannung} betrieben, die so groß ist, dass die Kugel gerade noch gehalten wird. Bei Gleichspannung bleiben leichte Kugeln wegen des Restmagnetismus trotz abgeschaltetem Strom länger hängen.
	\end{hilfebox}
	
	\begin{infobox}[title={\faIcon{info-circle} Die App phyphox}]

	\begin{hilfebox}[title={\faIcon{lightbulb} Zeitmessung mit phyphox}]
		Die Fallzeit wird mit der \textbf{akustischen Stoppuhr} in der phyphox-App bestimmt:
		\begin{enumerate}
			\item Öffnen Sie in phyphox das Experiment \textbf{„Akustische Stoppuhr"} (unter Zeitmessung).
			\item Im Reiter \textbf{„EINFACH"} stellen Sie die \textbf{Schwelle} so ein, dass Umgebungsgeräusche ignoriert werden.
			\item Stellen Sie eine \textbf{Mindestverzögerung} von ca. \SI{0,1}{\second} ein, damit das Startgeräusch die Messung nicht sofort stoppt.
			\item Das Klacken des Schalters startet die Stoppuhr, der Aufprall der Kugel stoppt sie.
		\end{enumerate}
		\textbf{Tipp:} Positionieren Sie das Smartphone auf halber Fallhöhe, um den Einfluss der Schallgeschwindigkeit zu minimieren.
	\end{hilfebox}

	\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 2: Auswertung}]
		Berechnen Sie aus den Messwerten die Erdbeschleunigung $g$. Die Beschleunigung wird wie folgt berechnet:
		
		\begin{equation*}
			h = \frac{1}{2} \cdot g \cdot t^2 \quad \Rightarrow \quad g = \frac{2 \cdot h}{t^2}
		\end{equation*}
		
		Tragen Sie die berechneten Werte in den Protokollbogen ein und berechnen Sie den \textbf{Mittelwert} sowie die \textbf{Messunsicherheit} von $g$.
	\end{aufgabe}
	
	\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 3: Vergleich mit Literaturwert}]
		Vergleichen Sie Ihren Messwert mit dem Literaturwert von $g_{\text{Lit}} = \SI{9,81}{\metre\per\second\squared}$, indem Sie die \textbf{relative Abweichung} bestimmen:
		
		\begin{equation*}
			\text{Relative Abweichung} = \frac{|g_{\text{Mess}} - g_{\text{Lit}}|}{g_{\text{Lit}}} \cdot 100\%
		\end{equation*}
	\end{aufgabe}

	\begin{infobox}[title={\faIcon{info-circle} Berechnung der Messunsicherheit}]
		\textbf{Mittelwert:} \quad $\displaystyle \bar{g} = \frac{1}{n} \sum_{i=1}^{n} g_i = \frac{g_1 + g_2 + \ldots + g_n}{n}$
		
		\hspace{1cm}\textit{Der Mittelwert ist der beste Schätzwert für den wahren Wert der Messgröße.}
		
		\vspace{0.3cm}
		\textbf{Standardabweichung:} \quad $\displaystyle s = \sqrt{\frac{1}{n-1} \sum_{i=1}^{n} (g_i - \bar{g})^2}$
		
		\hspace{1cm}\textit{Die Standardabweichung beschreibt die Streuung der Einzelmessungen um den Mittelwert.}
		
		\vspace{0.3cm}
		\textbf{Messunsicherheit des Mittelwerts:} \quad $\displaystyle \Delta \bar{g} = \frac{s}{\sqrt{n}}$
		
		\hspace{1cm}\textit{Die Messunsicherheit gibt an, in welchem Bereich der wahre Wert mit hoher}
		
		\hspace{1cm}\textit{Wahrscheinlichkeit (ca. 68\,\%) liegt. Sie wird kleiner, je mehr Messungen durchgeführt werden.}
	\end{infobox}
