	\end{center}

	% ============================================
	% Theoretischer Hintergrund
	% ============================================
	\section*{Theoretischer Hintergrund}
	
	Ein Fadenpendel schwingt bei kleinen Auslenkungen (Winkel $\varphi < 10°$) harmonisch. Die Schwingungsdauer $T$ hängt dabei nur von der \textbf{Pendellänge} $l$ und der \textbf{Erdbeschleunigung} $g$ ab -- nicht von der Masse oder der Amplitude!
	
	\begin{formelbox}[title={\faIcon{square-root-alt} Schwingungsdauer des Fadenpendels}]
		\[T = 2\pi \sqrt{\frac{l}{g}}\]
		
		\textbf{Dabei gilt:}
		\begin{itemize}
			\item $T$ -- Schwingungsdauer (Zeit für eine vollständige Hin- und Herbewegung) in s
			\item $l$ -- Pendellänge (Aufhängepunkt bis Schwerpunkt der Kugel) in m
			\item $g$ -- Erdbeschleunigung in $\si{\metre\per\second\squared}$
		\end{itemize}
	\end{formelbox}
	
	Durch Umstellen der Formel kann die Erdbeschleunigung $g$ bestimmt werden:
	
	\begin{formelbox}[title={\faIcon{square-root-alt} Erdbeschleunigung aus der Schwingungsdauer}]
		\[g = \frac{4\pi^2 \cdot l}{T^2}\]
	\end{formelbox}
	
	\begin{infobox}[title={\faIcon{info-circle} Vorteile dieser Methode}]
		\begin{itemize}
			\item \textbf{Längere Messzeiten:} Die Schwingungsdauer beträgt etwa 1--2 Sekunden, was die Zeitmessung erleichtert.
			\item \textbf{Mehrfachmessung:} Man misst die Zeit für $n$ Schwingungen und teilt durch $n$ → kleinerer relativer Fehler.
			\item \textbf{Kein Reaktionszeitfehler:} Die Stoppuhr kann beim Nulldurchgang gestartet und gestoppt werden.
		\end{itemize}
	\end{infobox}

	\begin{hilfebox}[title={\faIcon{lightbulb} Wichtige Hinweise}]
		\begin{itemize}
			\item Die \textbf{Pendellänge} $l$ wird vom Aufhängepunkt bis zum \textbf{Schwerpunkt} der Kugel gemessen (= Mittelpunkt bei einer homogenen Kugel).
			\item Der \textbf{Auslenkwinkel} sollte unter $10°$ bleiben, damit die Näherung für kleine Winkel gilt.
			\item Das Pendel soll in einer \textbf{Ebene} schwingen, nicht kreisen!
			\item Der Faden sollte möglichst \textbf{undehnbar} und leicht sein.
		\end{itemize}
	\end{hilfebox}

	\section*{Durchführung}
	
	Bauen Sie den Versuch gemäß Abbildung~\ref{fig:pendel} auf.
	
	\begin{enumerate}
		\item Messen Sie die Pendellänge $l$ vom Aufhängepunkt bis zur Mitte der Kugel.
		\item Lenken Sie das Pendel um einen kleinen Winkel ($\varphi < 10°$, ca. 5--10 cm bei $l \approx 1$ m) aus.
		\item Lassen Sie das Pendel los und warten Sie, bis die Schwingung gleichmäßig ist.
		\item Starten Sie die Stoppuhr beim Durchgang durch die Ruhelage.
		\item Messen Sie die Zeit $t_{10}$ für \textbf{10 vollständige Schwingungen}.
		\item Wiederholen Sie die Messung \textbf{dreimal} für jede Pendellänge.
		\item Führen Sie die Messung für \textbf{drei verschiedene Pendellängen} durch (z.\,B. $l = 0{,}50\,\text{m}$; $0{,}75\,\text{m}$; $1{,}00\,\text{m}$).
	\end{enumerate}

	\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 4: Grafische Auswertung}]
		Tragen Sie $T^2$ gegen $l$ auf. Bestimmen Sie die Steigung $m$ der Ausgleichsgeraden und berechnen Sie daraus $g$:
		\[T^2 = \frac{4\pi^2}{g} \cdot l \quad \Rightarrow \quad g = \frac{4\pi^2}{m}\]
	\end{aufgabe}

	\subsection*{Steigung und Erdbeschleunigung aus dem Diagramm}
	
	\textbf{Steigungsdreieck:}
	
	\vspace{0.3cm}
	
	$\Delta l = l_2 - l_1 = $ \rule{2cm}{0.4pt} $-$ \rule{2cm}{0.4pt} $=$ \rule{2cm}{0.4pt} m
	
	\vspace{0.3cm}
	
	$\Delta T^2 = T_2^2 - T_1^2 = $ \rule{2cm}{0.4pt} $-$ \rule{2cm}{0.4pt} $=$ \rule{2cm}{0.4pt} s$^2$
	
	\vspace{0.3cm}
	
	\textbf{Steigung:} $m = \dfrac{\Delta T^2}{\Delta l} = $ \rule{4cm}{0.4pt} $\dfrac{\text{s}^2}{\text{m}}$
	
	\vspace{0.5cm}
	
	\textbf{Erdbeschleunigung:} $g = \dfrac{4\pi^2}{m} = $ \rule{4cm}{0.4pt} $\dfrac{\text{m}}{\text{s}^2}$
