% Theoretischer Hintergrund

\section*{Theoretischer Hintergrund}

Ein Körper mit Trägheitsmoment $J$ schwingt an einer Drillachse (Richtgröße $D_r$) mit einer bestimmten Schwingungsdauer $T$. Aus der gemessenen Schwingungsdauer kann das Trägheitsmoment berechnet werden.

\begin{formelbox}[title={\faIcon{square-root-alt} Schwingungsdauer der Drillachse}]
	\[T = 2\pi \sqrt{\frac{J}{D_r}} \qquad\Longrightarrow\qquad J = \frac{D_r \cdot T^2}{4\pi^2}\]

	\textbf{Dabei gilt:}
	\begin{itemize}
		\item $T$ -- Schwingungsdauer (Zeit für eine vollständige Hin- und Herbewegung) in s
		\item $J$ -- Trägheitsmoment des schwingenden Körpers in $\si{kg.m^2}$
		\item $D_r$ -- Richtgröße (Federkonstante der Drillachse) in $\si{Nm}$
	\end{itemize}
\end{formelbox}

Auf dem Stab sitzen \textbf{zwei gleiche Massestücke} im Abstand $r$ von der Drehachse. Das Gesamt-Trägheitsmoment setzt sich zusammen aus:
\[
J = \underbrace{J_0}_{\text{Halter}} + \underbrace{2\,m\,r^2}_{\text{Massestücke}}
\]

\begin{infobox}[title={\faIcon{info-circle} Linearisierung -- Die Idee}]
	$J$ hängt \textbf{quadratisch} von $r$ ab -- wenn wir $J$ gegen $r$ auftragen, ergibt sich eine Parabel. Aber wenn wir $r^2$ als $x$-Achse wählen, wird daraus eine \textbf{Gerade}:
	\[
	\underbrace{J}_{y} = \underbrace{2m}_{\text{Steigung}\;k} \cdot \underbrace{r^2}_{x} + \underbrace{J_0}_{y\text{-Achsenabschnitt}}
	\]
	Aus der Steigung bestimmen wir die \textbf{Masse} $m$, aus dem $y$-Achsenabschnitt das \textbf{Eigenträgheitsmoment} $J_0$ des Halters.
\end{infobox}

\section*{Ziel}

Sie bestimmen aus einem $J$-$r^2$-Diagramm:
\begin{itemize}
	\item die \textbf{Masse} $m$ der Massestücke (aus der Steigung),
	\item das \textbf{Eigenträgheitsmoment} $J_0$ des Halters (aus dem $y$-Achsenabschnitt),
	\item eine \textbf{Fehlerabschätzung} für Ihre Ergebnisse (Extremgeraden-Methode).
\end{itemize}

\begin{hilfebox}[title={\faIcon{lightbulb} Tipps zur Messung}]
	\begin{itemize}
		\item Messen Sie die Zeit für \textbf{10 volle Schwingungen} ($10\,T$) -- das reduziert den Reaktionszeit-Fehler erheblich!
		\item Wiederholen Sie jede Messung \textbf{dreimal} und bilden Sie den Mittelwert.
		\item Den Abstand $r$ messen Sie von der \textbf{Mitte des Stabs} bis zur \textbf{Mitte des Massestücks}.
		\item Lenken Sie ca.\ $90°$ aus -- nicht zu weit, damit die Schwingung gleichmäßig bleibt.
	\end{itemize}
\end{hilfebox}

\section*{Durchführung}

\begin{enumerate}
	\item Lesen Sie die Richtgröße $D_r$ der Drillachse ab (Aufkleber oder gegeben).
	\item Stellen Sie die Massestücke symmetrisch auf den Abstand $r$ ein.
	\item Lenken Sie die Drillachse um ca.\ $90°$ aus und lassen Sie los.
	\item Messen Sie die Zeit für 10 volle Schwingungen. Wiederholen Sie dreimal.
	\item Wiederholen Sie für mindestens \textbf{5 verschiedene Abstände} $r$ (z.\,B.\ 4, 8, 12, 16, 20\,cm).
\end{enumerate}

\end{aufgabe}

\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 4: Steigung und $y$-Achsenabschnitt}]
	Bestimmen Sie mit einem \textbf{Steigungsdreieck} (zwei Punkte \textbf{auf der Geraden}, möglichst weit auseinander):
	\begin{itemize}
		\item die Steigung $k = \Delta J / \Delta r^2$
		\item daraus die Masse: $m = k/2$
		\item den $y$-Achsenabschnitt $J_0$
	\end{itemize}
	Vergleichen Sie $m$ mit dem Wert der Waage.
\end{aufgabe}

\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 5: Fehlerabschätzung}]
	\begin{enumerate}
		\item Berechnen Sie den relativen Fehler der Zeitmessung für Ihren kleinsten und größten Abstand.
		\item Zeichnen Sie Fehlerbalken ($\pm\Delta J$) mindestens am kleinsten und größten Messpunkt.
		\item Zeichnen Sie eine \textbf{steilste} und eine \textbf{flachste} Gerade durch die Fehlerbalken.
		\item Bestimmen Sie den Fehler der Masse: $\Delta m = (k_\text{max} - k_\text{min}) / 4$.
	\end{enumerate}
\end{aufgabe}

\begin{sternaufgabe}[title={\faIcon{star} Aufgabe 6: Reflexion}]
	\begin{enumerate}
		\item Bei welchem Abstand $r$ ist die Zeitmessung relativ am ungenauesten? Warum?
		\item Warum verdoppelt sich der relative Fehler beim Übergang von $T$ zu $J$? (Hinweis: $J \sim T^2$)
		\item Ist der pauschale Satz ``Die Messgeräte waren ungenau'' eine gute Fehleranalyse? Begründen Sie und formulieren Sie eine bessere.
	\end{enumerate}
\end{sternaufgabe}

\section*{Fehlerabschätzung}

\begin{infobox}[title={\faIcon{info-circle} Größtfehler-Methode (ohne Differenzialrechnung)}]
	Wenn eine Größe aus einer Formel berechnet wird, kann man den \textbf{größtmöglichen Fehler} abschätzen: Man setzt die ungünstigsten Werte der Messgrößen ein und schaut, wie stark sich das Ergebnis ändert.

	Faustregel: Da $J \sim T^2$ gilt, verdoppelt sich der relative Fehler:
	\[\frac{\Delta J}{J} \approx 2 \cdot \frac{\Delta T}{T}\]
\end{infobox}

\subsection*{Extremgeraden-Methode}

\begin{enumerate}
	\item Zeichnen Sie an jeden Messpunkt einen \textbf{vertikalen Fehlerbalken} ($\pm\Delta J$, nutzen Sie $\Delta J / J \approx 3\,\%$).
	\item Zeichnen Sie im Diagramm zusätzlich:
	\begin{itemize}
		\item \textcolor{red}{\textbf{rote Gerade}}: \textbf{steilste} Gerade, die noch durch alle Fehlerbalken passt
		\item \textcolor{blue}{\textbf{blaue Gerade}}: \textbf{flachste} Gerade, die noch durch alle Fehlerbalken passt
	\end{itemize}
	\item Bestimmen Sie die Steigungen:
\end{enumerate}

\vspace{0.3cm}
$k_\text{max} = $ \rule{3cm}{0.4pt} kg \qquad $k_\text{min} = $ \rule{3cm}{0.4pt} kg
