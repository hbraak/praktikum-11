	\begin{center}
		{\LARGE\bfseries Herleitung der Pendelgleichung}\\[0.3cm]
		{\large Warum gilt $T = 2\pi\sqrt{l/g}$\,?}
	\end{center}
	\begin{minipage}[t]{0.58\textwidth}
		
		\begin{schrittbox}[title={\faIcon{shoe-prints} Schritt 1: Rücktreibende Kraft}]
			Bei kleiner Auslenkung $s$ aus der Ruhelage wirkt auf die Pendelmasse die Komponente der Gewichtskraft:
			\[F = -mg \cdot \sin\varphi \approx -mg \cdot \frac{s}{l}\]
			Das Minuszeichen zeigt: Die Kraft zeigt immer zur Ruhelage zurück.
		\end{schrittbox}
		
		\vspace{0.2cm}
		
		\begin{schrittbox}[title={\faIcon{shoe-prints} Schritt 2: Beschleunigung (Newton)}]
			Mit dem 2. Newton'schen Gesetz $F = m \cdot a$:
			\[m \cdot a = -mg \cdot \frac{s}{l}\]
			Die Masse $m$ kürzt sich heraus:
			\[\boxed{a = -\frac{g}{l} \cdot s}\]
		\end{schrittbox}
		
		\vspace{0.2cm}
		
		\begin{schrittbox}[title={\faIcon{shoe-prints} Schritt 3: Harmonische Schwingung erkennen}]
			Die Beschleunigung ist \textbf{proportional zur Auslenkung} und \textbf{entgegengerichtet} (Minuszeichen). Das ist die Bedingung für eine \textbf{harmonische Schwingung}.
			
			Allgemein gilt für harmonische Schwingungen:
			\[s(t) = s_0 \cdot \sin\left(\frac{2\pi}{T} \cdot t\right)\]
			
			Die Beschleunigung ist die zweite Ableitung des Weges:
			\[a(t) = \ddot{s}(t) = -s_0 \cdot \left(\frac{2\pi}{T}\right)^2 \cdot \sin\left(\frac{2\pi}{T} \cdot t\right) = -\frac{4\pi^2}{T^2} \cdot s(t)\]
			
			Daraus folgt der Zusammenhang:
			\[\boxed{\frac{a}{s} = -\frac{4\pi^2}{T^2}}\]
		\end{schrittbox}
		
		\vspace{0.2cm}
		
		\begin{schrittbox}[title={\faIcon{shoe-prints} Schritt 4: Gleichsetzen und Auflösen}]
			\[-\frac{g}{l} = -\frac{4\pi^2}{T^2}\]
			Nach $T^2$ auflösen:
			\[T^2 = \frac{4\pi^2 \cdot l}{g}\]
			Wurzel ziehen:
			\[\boxed{T = 2\pi\sqrt{\frac{l}{g}}}\]
		\end{schrittbox}
		
	\end{minipage}

		\begin{infobox}[title={\faIcon{info-circle} Kleinwinkelnäherung}]
			Für kleine Winkel gilt:
			\[\sin\varphi \approx \varphi = \frac{s}{l}\]
			(Winkel im Bogenmaß!)
			
			Gültig für $\varphi < 10°$.
		\end{infobox}
