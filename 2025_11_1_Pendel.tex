	\section*{Theoretischer Hintergrund}
	
	Eine schwere Kugel hängt an einem sehr langen Faden und wird ausgelenkt. Da sie dabei etwas angehoben wird, erhält sie \textbf{Lageenergie} (potentielle Energie). Diese wandelt sich beim Schwingen -- wie bei einer Schaukel -- in \textbf{Bewegungsenergie} (kinetische Energie) und dann wieder in Lageenergie um. Bleiben die Geschwindigkeit und der Luftwiderstand gering, so erfolgt eine fast verlustfreie Umwandlung der Energien.
	
	\vspace{0.5cm}

	\vspace{0.3cm}
	
	Die Lageenergie wird im tiefsten Punkt fast vollständig in Bewegungsenergie umgewandelt. Mit Hilfe der Energiebilanz kann man die \textbf{Dunkelzeit} berechnen, während der die im tiefsten Punkt angebrachte Lichtschranke vom Kugeldurchmesser abgedunkelt wird.
	
	\begin{formelbox}[title={\faIcon{square-root-alt} Herleitung der Dunkelzeit}]
		\textbf{Energieerhaltung:} \quad $E_{\text{pot}} = E_{\text{kin}}$
		
		\vspace{0.2cm}
		\[m \cdot g \cdot h = \frac{1}{2} m \cdot v^2 \quad \Rightarrow \quad v = \sqrt{2 \cdot g \cdot h}\]
		
		\vspace{0.3cm}
		\textbf{Dunkelzeit:} Die Kugel durchquert die Lichtschranke mit der Geschwindigkeit $v$ und legt dabei den Weg $d_K$ (Kugeldurchmesser) zurück:
		
		\vspace{0.2cm}
		\[t_D = \frac{d_K}{v} = \frac{d_K}{\sqrt{2 \cdot g \cdot h}}\qquad\textbf{(Gleichung 1)}\] 
	\end{formelbox}
	
	%	\begin{hilfebox}
		%		Die Masse $m$ kürzt sich bei der Energiebilanz heraus! Die Dunkelzeit ist daher \textbf{unabhängig von der Kugelmasse} -- nur der Kugeldurchmesser und die Auslenkhöhe bestimmen das Ergebnis.
		%	\end{hilfebox}

	\section*{Versuchsaufbau und Durchführung}
	
	\begin{materialbox}
		\begin{itemize}
			\item Holzkugel (rot, $d_K = \SI{7}{cm}$)
			\item Eisenkugel (etwas geringerer Durchmesser)
			\item Faden (möglichst lang)
			\item Lichtschranke mit Zählgerät
			\item Stativmaterial
			\item Messschieber / Lineal
		\end{itemize}
	\end{materialbox}
	
	\begin{hilfebox}[title={\faIcon{lightbulb} Durchführung}]
		\begin{enumerate}
			\item Hängen Sie die Holzkugel an einem Faden auf.
			\item Ordnen Sie die Lichtschranke im tiefsten Punkt so an, dass der \textbf{volle Kugeldurchmesser} die Lichtschranke unterbricht.
			\item Schalten Sie das Zählgerät so, dass die \textbf{Dunkelzeit} registriert wird (Brücke oben).
			\item Messen Sie für verschiedene Auslenkhöhen $h$ die Dunkelzeit $t_D$.
			\item Wiederholen Sie eine Messung mit der Eisenkugel.
		\end{enumerate}
	\end{hilfebox}

	\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 6: Massenunabhängigkeit}]

	\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 7: Fehleranalyse}]
		Berechnen Sie zu jeder Messung den \textbf{relativen Fehler}, bezogen auf den theoretischen Wert:

			\text{Relativer Fehler} = \frac{|t_{D,\text{Mess}} - t_{D,\text{Theo}}|}{t_{D,\text{Theo}}} \cdot 100\%
		\end{equation*}
	\end{aufgabe}
