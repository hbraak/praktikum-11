	\section*{Theoretischer Hintergrund}
	
	Bei der \textbf{Elektrolyse} wird Wasser mit elektrischer Energie in Wasserstoff und Sauerstoff zerlegt:
	\[2\,\text{H}_2\text{O} \longrightarrow 2\,\text{H}_2 + \text{O}_2\]
	
	Der \textbf{energetische Wirkungsgrad} $\eta_E$ beschreibt, welcher Anteil der elektrischen Energie als chemische Energie im Wasserstoff gespeichert wird:
	
	\begin{formelbox}[title={\faIcon{square-root-alt} Energetischer Wirkungsgrad}]
		\[\eta_E = \frac{E_{H_2}}{E_{\text{el}}} = \frac{V_{H_2} \cdot H_O}{U \cdot I \cdot t}\]
		
		\begin{tabular}{ll}
			$V_{H_2}$ -- Wasserstoffvolumen in m$^3$ & $H_O = \SI{12,745e6}{\joule\per\cubic\metre}$ (Brennwert) \\
			$U$ -- Spannung in V & $I$ -- Stromstärke in A \\
			$t$ -- Zeit in s &
		\end{tabular}
	\end{formelbox}
	
	\begin{hilfebox}[title={\faIcon{lightbulb} Doppelelektrolyseur}]
		Unser Elektrolyseur besteht aus \textbf{zwei Zellen in Reihe}. Beide produzieren Wasserstoff!
		
		$\Rightarrow$ Für $\eta_E$ das \textbf{Gesamtvolumen} $V_{\text{ges}}$ verwenden.
	\end{hilfebox}

	\section*{Aufgaben}
	
	\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 1: Messung}]
		Messen Sie alle \SI{2}{min} Spannung $U$, Stromstärke $I$ und Wasserstoffvolumen $V$.
		
		Führen Sie 7 Messungen durch (bis $t = \SI{14}{min} = \SI{840}{s}$).
	\end{aufgabe}
	
	\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 2: Diagramm}]
		Tragen Sie das Volumen $V$ gegen die Zeit $t$ auf und zeichnen Sie eine Ausgleichsgerade.
	\end{aufgabe}
	
	\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 3: Wirkungsgrad berechnen}]
		Berechnen Sie den energetischen Wirkungsgrad $\eta_E$ mit:
		\begin{itemize}
			\item Mittelwerten $\overline{U}$ und $\overline{I}$
			\item Gesamtvolumen $V_{\text{ges}}$ (letzter Messwert)
			\item Gesamtzeit $t = \SI{840}{s}$
		\end{itemize}
		Vergleichen Sie mit dem Literaturwert: $\eta_E = 70$--$85\,\%$
	\end{aufgabe}
	
	\begin{sternaufgabe}[title={\faIcon{star} Aufgabe 4: Offset-Korrektur}]
		Die Trendlinie schneidet die $t$-Achse nicht bei $t=0$, sondern bei $t_{\text{off}}$.
		
		\textbf{a)} Bestimmen Sie $t_{\text{off}}$ aus dem Diagramm.
		
		\textbf{b)} Berechnen Sie $\eta_E$ erneut mit der effektiven Zeit $t_{\text{eff}} = t - t_{\text{off}}$.
		
		\textbf{c)} Erklären Sie die Ursache des Offsets.
	\end{sternaufgabe}
