	\section*{Theoretischer Hintergrund}
	
	Wasserstoff gilt als wichtiger Energieträger der Zukunft. Er kann durch \textbf{Elektrolyse} aus Wasser gewonnen werden -- dabei wird elektrische Energie in chemische Energie umgewandelt. Die Umkehrreaktion findet in einer \textbf{Brennstoffzelle} statt.
	
	\begin{center}
		\begin{tikzpicture}[scale=1.0, every node/.style={font=\small}]
			% Linker Pfeil (Elektrische Energie)
			\fill[black] (0,0.5) -- (0,-0.5) -- (2,0) -- cycle;
			\node[above] at (1,0.7) {$U \cdot I \cdot t$};
			\node[below] at (1,-0.9) {Elektrische Energie};
			
			% Kasten (Elektrolyseur)
			\draw[thick] (2.2,-0.8) rectangle (5.8,0.8);
			\node[align=center] at (4,0) {\textbf{Elektro-}\\\textbf{lyseur}};
			
			% Rechter Pfeil (Chemische Energie)
			\fill[black] (6,0.5) -- (6,-0.5) -- (8,0) -- cycle;
			\node[above] at (7,0.7) {$V_{H_2} \cdot H_O$};
			\node[below] at (7,-0.9) {Chemische Energie};
		\end{tikzpicture}
	\end{center}
	
	Bei einem \textbf{PEM-Elektrolyseur} (Proton Exchange Membrane) trennt eine protonenleitende Polymermembran die beiden Elektroden:
	
	\begin{center}
		$\mathbf{2\,\text{H}_2\text{O} \longrightarrow 2\,\text{H}_2 + \text{O}_2}$
	\end{center}
	
	\begin{formelbox}[title={\faIcon{square-root-alt} Energetischer Wirkungsgrad $\eta_E$}]
		Der energetische Wirkungsgrad beschreibt das Verhältnis der im Wasserstoff gespeicherten chemischen Energie zur aufgewandten elektrischen Energie:
		\[\eta_E = \frac{E_{H_2}}{E_{\text{el}}} = \frac{V_{H_2} \cdot H_O}{U \cdot I \cdot t}\]
		
		\textbf{Dabei gilt:}
		\begin{itemize}
			\item $V_{H_2}$ -- Volumen des erzeugten Wasserstoffs in m$^3$
			\item $H_O$ -- Brennwert von Wasserstoff: $H_O = \SI{12,745e6}{\joule\per\cubic\metre}$
			\item $U$ -- Spannung in V, \quad $I$ -- Stromstärke in A, \quad $t$ -- Zeit in s
		\end{itemize}
	\end{formelbox}
	
	\begin{hilfebox}[title={\faIcon{lightbulb} Hinweis zum Doppelelektrolyseur}]
		Der verwendete HTEC-Elektrolyseur besteht aus \textbf{zwei in Reihe geschalteten PEM-Zellen}:
		\begin{itemize}
			\item Der Strom $I$ ist in beiden Zellen gleich
			\item Die Spannung teilt sich auf: $U_{\text{ges}} \approx 2 \cdot U_{\text{Zelle}}$
			\item \textbf{Beide} Zellen produzieren Wasserstoff!
		\end{itemize}
		
		$\Rightarrow$ Für $\eta_E$ das \textbf{Gesamtvolumen} $V_{\text{ges}}$ verwenden.
	\end{hilfebox}

	\section*{Aufgaben}
	
	\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 1: Messung}]
		\begin{enumerate}
			\item Füllen Sie den Elektrolyseur mit destilliertem Wasser und lassen Sie ihn \SI{2}{min} im Leerlauf laufen.
			\item Stellen Sie eine Spannung sodass sich ein Strom von ca. \SI{150}{mA} einstellt.
			\item Starten Sie die Zeitmessung und notieren Sie alle \SI{2}{min} die Werte für Spannung $U$, Stromstärke $I$ und Wasserstoffvolumen $V$.
			\item Führen Sie 7 Messungen durch (bis $t = \SI{14}{min}$).
		\end{enumerate}
	\end{aufgabe}
	
	\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 2: $t$-$V$-Diagramm}]
		Tragen Sie das gemessene Volumen $V$ als Funktion der Zeit $t$ in ein Diagramm ein. Zeichnen Sie eine Ausgleichsgerade (Trendlinie) durch die Messwerte.
	\end{aufgabe}
	
	\begin{aufgabe}[title={\faIcon{pencil-alt} Aufgabe 3: Wirkungsgrad berechnen}]
		Berechnen Sie den energetischen Wirkungsgrad $\eta_E$ mit:
		\begin{itemize}
			\item Mittelwerten $\overline{U}$ und $\overline{I}$
			\item Gesamtvolumen $V_{\text{ges}}$ (letzter Messwert)
			\item Gesamtzeit $t = \SI{840}{s}$
		\end{itemize}
		Vergleichen Sie mit dem Literaturwert: $\eta_E = 70$--$85\,\%$
	\end{aufgabe}
	
	\begin{sternaufgabe}[title={\faIcon{star} Aufgabe 4: Offset-Korrektur}]
		Die Trendlinie schneidet die $t$-Achse nicht bei $t=0$, sondern bei $t_{\text{off}}$.
		
		\textbf{a)} Bestimmen Sie $t_{\text{off}}$ aus dem Diagramm.
		
		\textbf{b)} Berechnen Sie $\eta_E$ erneut mit der effektiven Zeit $t_{\text{eff}} = t - t_{\text{off}}$.
		
		\textbf{c)} Erklären Sie die Ursache des Offsets.
	\end{sternaufgabe}
